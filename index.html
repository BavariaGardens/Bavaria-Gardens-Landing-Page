    <script>

        const bookingModal = document.getElementById('bookingModal');

        const bookNowBtn = document.getElementById('bookNowBtn');

        const bookNowFooterBtn = document.getElementById('bookNowFooterBtn');

        const closeModalBtn = document.getElementById('closeModalBtn');


        function openModal() {

            bookingModal.style.display = 'flex';

            setTimeout(() => {

                bookingModal.querySelector('div').style.transform = 'scale(1)';

            }, 10);

        }


        function closeModal() {

            bookingModal.querySelector('div').style.transform = 'scale(0.95)';

            setTimeout(() => {

                bookingModal.style.display = 'none';

            }, 300);

        }


        bookNowBtn.addEventListener('click', openModal);

        bookNowFooterBtn.addEventListener('click', openModal);

        closeModalBtn.addEventListener('click', closeModal);

        window.addEventListener('click', (event) => {

            if (event.target === bookingModal) {

                closeModal();

            }

        });

    </script>
